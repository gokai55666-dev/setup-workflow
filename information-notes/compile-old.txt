Draft: .github/workflows/learn-github-actions.yml
name: Learn GitHub Actions

# Trigger on push to main or on workflow dispatch for manual testing
on:
  push:
    branches: [ main ]
    paths:
      - '**/*.js'
      - '**/*.py'
      - '.github/workflows/learn-github-actions.yml'
  workflow_dispatch:

permissions:
  contents: read
  actions: read

jobs:
  setup:
    name: Setup environment
    runs-on: ubuntu-latest

    steps:
      # Checkout repository
      - name: Checkout code
        uses: actions/checkout@v6

      # Setup Node.js (if applicable)
      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      # Setup Python (if applicable)
      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'

      - name: Display versions
        run: |
          node --version || echo "Node not installed"
          python --version || echo "Python not installed"

  test:
    name: Run tests
    runs-on: ubuntu-latest
    needs: setup

    steps:
      - uses: actions/checkout@v6

      # Cache dependencies
      - name: Cache npm modules
        uses: actions/cache@v3
        with:
          path: ~/.npm
          key: ${{ runner.os }}-node-${{ hashFiles('**/package-lock.json') }}
          restore-keys: |
            ${{ runner.os }}-node-

      # Install dependencies
      - name: Install dependencies
        run: npm install

      # Run tests (adjust as needed)
      - name: Run tests
        run: npm test

  lint:
    name: Lint code
    runs-on: ubuntu-latest
    needs: setup

    steps:
      - uses: actions/checkout@v6

      - name: Install lint tools
        run: |
          npm install eslint || echo "No ESLint"
          pip install flake8 || echo "No flake8"

      - name: Run linter
        run: |
          npm run lint || echo "No JS lint"
          flake8 . || echo "No Python lint"

  notify:
    name: Notification (optional)
    runs-on: ubuntu-latest
    needs: [test, lint]
    if: failure()

    steps:
      - name: Notify on failure
        run: echo "One or more jobs failed in learn-github-actions workflow."

Explicit permissions: Minimum required (contents: read, actions: read) to follow the least-privilege principle.
Modular jobs: Separates setup, test, lint; allows better error isolation.
Caching: Speeds up builds, especially with npm or other dependencies.
Notification stage: Detect and report failures across jobs.
